Carousel Indicators Demo

In this example, we use Bootstrap's carousel indicators (require jQuery 1.9.1 or higher)

<script src="//code.jquery.com/jquery-1.9.1.js"></script>
<script src="jquery.backstretch.min.js"></script>
<script>
$("#indicator-0").addClass("active");
$(".carousel-example").backstretch(["pot-holder.jpg", "coffee.jpg", "dome.jpg"]);

// Listen to the backstretch event to highlight the current indicator
$(window).on("backstretch.before", function (e, instance, index) {
	jQuery(".carousel-indicators li").removeClass("active");
	$("#indicator-"+index.toString()).addClass("active")
});

// This is optional if you want the backstretch pause when user hover over the indicators
$(".carousel-indicators").mouseover(function(){
	$(".carousel-example").backstretch("pause");
	$(".indicator").each(function(index){
		$(this).click(function(){
			$(".carousel-indicators li").removeClass("active");
			$(".carousel-example").backstretch("show", index);
			$(this).addClass("active");
		});
	});
});

$(".carousel-indicators").mouseleave(function(){
	$(".carousel-example").backstretch("resume");
	// Listen to the backstretch event to highlight the current indicator
	$(window).on("backstretch.before", function (e, instance, index) {
		$(".carousel-indicators li").removeClass("active");
		$("#indicator-"+index.toString()).addClass("active")
	});
});
</script>
